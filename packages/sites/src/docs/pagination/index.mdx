## 单页隐藏

只有一页的时候 通过hideOnSinglePage属性隐藏分页器


<DemoBlock>
    <HideOnSinglePage />

    ```tsx
    import {Checkbox, Pagination} from '@g-ui/components'
    import {useState} from "react";

    const Demo = () => {
        const [checked, setChecked] = useState(false)
        return (
            <>
                <Checkbox checked={checked}
                    onChange={(event) => setChecked(event.target.checked)}
                >
                    hideOnSinglePage
                </Checkbox>
                <br />
                <br />
                <Pagination hideOnSinglePage={checked}/>
            </>
        )
    }

    export default Demo
    ```
</DemoBlock>


## 受控组件

就是react的受控组件


<DemoBlock>

    <Controlled/>

    ```tsx
    import {Pagination} from '@g-ui/components'
    import {useState} from "react";

    const Demo = () => {
        const [v, setV] = useState(1)
        const handleChange = (newCurrent: number) => {
            setV(newCurrent)
        }
        return (
            <>
                当前页面: {v}
                <br/>
                <br/>
                <Pagination total={1000} current={v} onChange={handleChange} showQuickJumper/>
                <br/>
                <br/>
                <Pagination total={1000} current={v} onChange={handleChange} simple/>
            </>
        )
    }
    export default Demo
    ```
</DemoBlock>
